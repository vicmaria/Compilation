sig
  type identifier = string
  type expression =
      EConst of MJ.constant
    | EGetVar of MJ.identifier
    | EUnOp of MJ.unop * MJ.expression
    | EBinOp of MJ.binop * MJ.expression * MJ.expression
    | EMethodCall of MJ.expression * MJ.identifier * MJ.expression list
    | EArrayGet of MJ.expression * MJ.expression
    | EArrayAlloc of MJ.expression
    | EArrayLength of MJ.expression
    | EThis
    | EObjectAlloc of MJ.identifier
  and constant = LMJ.constant = ConstBool of bool | ConstInt of int32
  and binop = LMJ.binop = OpAdd | OpSub | OpMul | OpLt | OpAnd
  and unop = LMJ.unop = UOpNot
  and instruction =
      IBlock of MJ.instruction list
    | IIf of MJ.expression * MJ.instruction * MJ.instruction
    | IWhile of MJ.expression * MJ.instruction
    | ISyso of MJ.expression
    | ISetVar of MJ.identifier * MJ.expression
    | IArraySet of MJ.identifier * MJ.expression * MJ.expression
  and typ = TypInt | TypBool | TypIntArray | Typ of MJ.identifier
  and metho = {
    formals : (MJ.identifier * MJ.typ) list;
    result : MJ.typ;
    locals : (MJ.identifier * MJ.typ) list;
    body : MJ.instruction list;
    return : MJ.expression;
  }
  and clas = {
    extends : MJ.identifier option;
    attributes : (MJ.identifier * MJ.typ) list;
    methods : (MJ.identifier * MJ.metho) list;
  }
  and program = {
    name : MJ.identifier;
    defs : (MJ.identifier * MJ.clas) list;
    main_args : MJ.identifier;
    main : MJ.instruction;
  }
end
