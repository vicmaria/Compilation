Module Print

module Print: sig .. end

Provides helper functions to print on an output channel. Some functions use a current level of indentation which starts at 0 and can be updated by adding an offset to it.

val nl : Stdlib.out_channel -> unit

nl out prints on the output channel out a newline and then the current level of indentation.

val indent : int -> (Stdlib.out_channel -> 'a -> unit) -> Stdlib.out_channel -> 'a -> unit

indent offset producer out x prints on the output channel out the object x using producer. The offset is added to the current level of indentation. A newline and the new level of indentation is printed before the object x.

val indent_t : int -> (Stdlib.out_channel -> unit) -> Stdlib.out_channel -> unit

indent_t offset producer out prints on the output channel out using producer. The offset is added to the current level of indentation. A newline and the new level of indentation is printed before using the producer.

val list : (Stdlib.out_channel -> 'a -> unit) -> Stdlib.out_channel -> 'a list -> unit

list elem out xs prints on the output channel out the list xs using elem to print each element.

val prec_list : (Stdlib.out_channel -> unit) ->
       (Stdlib.out_channel -> 'a -> unit) -> Stdlib.out_channel -> 'a list -> unit

prec_list delim elem out xs prints on the output channel out the list xs using elem to print each element. Each element of the list is preceded by a delimiter which is printed with delim.

val term_list : (Stdlib.out_channel -> unit) ->
       (Stdlib.out_channel -> 'a -> unit) -> Stdlib.out_channel -> 'a list -> unit

term_list delim elem out xs prints on the output channel out the list xs using elem to print each element. Each element of the list is terminated by a delimiter which is printed with delim.

val sep_list : (Stdlib.out_channel -> unit) ->
       (Stdlib.out_channel -> 'a -> unit) -> Stdlib.out_channel -> 'a list -> unit

sep_list delim elem out xs prints on the output channel out the list xs using elem to print each element. Each element of the list is separated by a delimiter which is printed with delim.

val space : Stdlib.out_channel -> unit

space out prints a whitespace on the output channel out.

val comma : Stdlib.out_channel -> unit

comma out prints a comma on the output channel out.

val semicolon : Stdlib.out_channel -> unit

semicolon out prints a semicolon on the output channel out.

val print_string : Stdlib.out_channel -> string -> unit

print_string out s prints the string s on the output channel out.
